HTTP

POST /login
- body: {
    username: string,
    password: string
  }
- return on success: {
    success: true
  }
- return on failure: {
    success: false,
    reason: <
      already-logged-in,
      json-loading-failed,
      empty-username,
      empty-password,
      db-error,
      wrong-credentials
    >
  }


FROM CLIENT TO SERVER

create-game
- argument: {
    name: string,
    maxPlayers: int,
    password: string [optional]
  }
- return on success: {
    success: true,
    id: game_id
  }
- return on failure: {
    success: false,
    reason: <
      not-logged-in,
      invalid-arguments,
      already-in-game,
      invalid-game-name,
      invalid-max-players,
      invalid-game-password,
      insert-failed
    >
  }


join-game
- argument: {
    id: int,
    password: string [optional]
  }
- return on success: {
    success: true,
    players: [list of players]
  }
- return on failure: {
    success: false,
    reason: <
      not-logged-in,
      already-in-game,
      invalid-arguments,
      invalid-game-id,
      db-error,
      game-started,
      game-full,
      password-required,
      invalid-password
    >
  }

start-game
- arguments: none
- return on success: {
    success: true
  }
- return on failure: {
    success: false,
    reason: <
      not-logged-in,
      invalid-arguments,
      not-in-game,
      db-error,
      game-already-started,
      invalid-game-id,
      not-creator
    >
  }

leave-game
- arguments: none
- return on success: {
    success: true
  }
- return on failure: {
    success: false,
    reason: <
      not-logged-in,
      invalid-arguments,
      not-in-game,
      db-error
    >
  }
